35  package org.owasp.encoder.esapi;
36  
37  import java.io.IOException;
38  import org.owasp.encoder.Encode;
39  import org.owasp.esapi.Encoder;
40  import org.owasp.esapi.codecs.Codec;
41  import org.owasp.esapi.errors.EncodingException;
42  import org.owasp.esapi.reference.DefaultEncoder;
43  
44  /**
45   * ESAPIEncoder is a singleton implementation of the ESAPI Encoder API.  It
46   * is meant to allow quick and easy drop-in replacement of the default
47   * encoder included with the ESAPI library, as the Encoder libraries are
48   * faster and use less memory thus cause fewer garbage collections.
49   *
50   * <p>Please note that the OWASP Java Encoders does not implement all
51   * the encodings of the ESAPI Encoder API.  In such situations this
52   * implementation will fallback onto the default reference implementation
53   * included with ESAPI.  Thus you should see the performance benefit from
54   * the methods included in the Encoder, but still maintain compatibility
55   * with all methods from ESAPI Encoder.</p>
56   *
57   * <p>For clarity, the reason the OWASP Java Encoders do not include some
58   * of the ESAPI library is that the Encoders library is specifically focused
59   * on <i>encoding</i>, and thus does not include:</p>
60   *
61   * <ul>
62   *     <li>Input validation/normalization methods:
63   *     {@link org.owasp.esapi.Encoder#canonicalize(String)},
64   *     {@link org.owasp.esapi.Encoder#canonicalize(String, boolean)},
65   *     {@link org.owasp.esapi.Encoder#canonicalize(String, boolean, boolean)}</li>
66   *
67   *     <li>Decoding methods:
68   *     {@link org.owasp.esapi.Encoder#decodeForHTML(String)},
69   *     {@link org.owasp.esapi.Encoder#decodeFromURL(String)}</li>
70   *
71   *     <li>Binary-to-text/text-to-binary:
72   *     {@link org.owasp.esapi.Encoder#encodeForBase64(byte[], boolean)},
73   *     {@link org.owasp.esapi.Encoder#decodeFromBase64(String)}.</li>
74   *
75   *     <li>Bind-able APIs (such as {@link java.sql.PreparedStatement}:
76   *     {@link org.owasp.esapi.Encoder#encodeForSQL(org.owasp.esapi.codecs.Codec, String)},
77   *     {@link org.owasp.esapi.Encoder#encodeForXPath(String)},
78   *     {@link org.owasp.esapi.Encoder#encodeForOS(org.owasp.esapi.codecs.Codec, String)}</li>
79   *
80   *     <li>Rarely-used or alternate compatible encoding:
81   *     {@link org.owasp.esapi.Encoder#encodeForVBScript(String)},
82   *     {@link org.owasp.esapi.Encoder#encodeForLDAP(String)},
83   *     {@link org.owasp.esapi.Encoder#encodeForDN(String)}</li>
84   * </ul>
85   *
86   * <p>(Please note that with sufficient feedback from the user base, the above
87   * mentioned methods may be implemented in future releases of the OWASP
88   * Java Encoders, if/when that happens, this shim class will be updated to
89   * call out to the new methods.)</p>
90   *
91   * <p>You may notice that this class does not actually implement Encoder
92   * itself.  Instead it simply provides a {@link #getInstance()} method that
93   * does.  This allows the implementation details maximum flexibility by not
94   * creating a any public API that would restrict changes later</p>
95   *
96   * @author jeffi
97   */
98  public final class ESAPIEncoder {
99  
100     /** No instances. */
101     private ESAPIEncoder() {}
102 
103     /**
104      * Returns an instance of the Encoder.  This method is the only supported
105      * mechanism by which an ESAPIEncoder instance should be obtained.  The
106      * returned implementation is guaranteed to be thread-safe for the methods
107      * that the OWASP Java Encoders implement (see class documentation).
108      * Though not a requirement of the ESAPI Encoder API, the returned value
109      * is also serializable.
110      *
111      * @return An encoder implementation that uses the OWASP Java Encoders
112      * for most of the common encoding methods.
113      */
114     public static Encoder getInstance() {
115         return Impl.INSTANCE;
116     }
117 
118     /**
119      * This is the private singleton that implements the ESAPI Encoder shim.
120      * It is implemented as a single-value enum to get all the "free" singleton
121      * properties associated with enums--such as serialization, and on-demand
122      * initialization.
123      *
124      * <p>The implementation is intentionally private to avoid any API baggage.
125      * The instance should be obtained using
126      * {@link org.owasp.encoder.esapi.ESAPIEncoder#getInstance()}.</p>
127      */
128     private enum Impl implements Encoder {
129         /**
130          * The singleton instance.
131          */
132         INSTANCE;
133 
134         /**
135          * The reference encoder from ESAPI.  Any ESAPI method without an
136          * OWASP Java Encoder equivalent is delegated to this instance.
137          */
138         private final Encoder _referenceEncoder = DefaultEncoder.getInstance();
139 
140         /** {@inheritDoc} */
141         public String canonicalize(String s) {
142             return _referenceEncoder.canonicalize(s);
143         }
144 
145         /** {@inheritDoc} */
146         public String canonicalize(String s, boolean strict) {
147             return _referenceEncoder.canonicalize(s, strict);
148         }
149 
150         /** {@inheritDoc} */
151         public String canonicalize(String s, boolean restrictMultiple, boolean restrictMixed) {
152             return _referenceEncoder.canonicalize(s, restrictMultiple, restrictMixed);
153         }
154 
155         /** {@inheritDoc} */
156         public String encodeForCSS(String s) {
157             return Encode.forCssString(s);
158         }
159 
160         /** {@inheritDoc} */
161         public String encodeForHTML(String s) {
162             return Encode.forHtml(s);
163         }
164 
165         /** {@inheritDoc} */
166         public String decodeForHTML(String s) {
167             return _referenceEncoder.decodeForHTML(s);
168         }
169 
170         /** {@inheritDoc} */
171         public String encodeForHTMLAttribute(String s) {
172             return Encode.forHtmlAttribute(s);
173         }
174 
175         /** {@inheritDoc} */
176         public String encodeForJavaScript(String s) {
177             return Encode.forJavaScript(s);
178         }
179 
180         /** {@inheritDoc} */
181         public String encodeForVBScript(String s) {
182             return _referenceEncoder.encodeForVBScript(s);
183         }
184 
185         /** {@inheritDoc} */
186         public String encodeForSQL(Codec codec, String s) {
187             return _referenceEncoder.encodeForSQL(codec, s);
188         }
189 
190         /** {@inheritDoc} */
191         public String encodeForOS(Codec codec, String s) {
192             return _referenceEncoder.encodeForOS(codec, s);
193         }
194 
195         /** {@inheritDoc} */
196         public String encodeForLDAP(String s) {
197             return _referenceEncoder.encodeForLDAP(s);
198         }
199 
200         /** {@inheritDoc} */
201         public String encodeForDN(String s) {
202             return _referenceEncoder.encodeForDN(s);
203         }
204 
205         /** {@inheritDoc} */
206         public String encodeForXPath(String s) {
207             return _referenceEncoder.encodeForXPath(s);
208         }
209 
210         /** {@inheritDoc} */
211         public String encodeForXML(String s) {
212             return Encode.forXml(s);
213         }
214 
215         /** {@inheritDoc} */
216         public String encodeForXMLAttribute(String s) {
217             return Encode.forXmlAttribute(s);
218         }
219 
220         /** {@inheritDoc} */
221         public String encodeForURL(String s) throws EncodingException {
222             return Encode.forUri(s);
223         }
224 
225         /** {@inheritDoc} */
226         public String decodeFromURL(String s) throws EncodingException {
227             return _referenceEncoder.decodeFromURL(s);
228         }
229 
230         /** {@inheritDoc} */
231         public String encodeForBase64(byte[] bytes, boolean wrap) {
232             return _referenceEncoder.encodeForBase64(bytes, wrap);
233         }
234 
235         /** {@inheritDoc} */
236         public byte[] decodeFromBase64(String s) throws IOException {
237             return _referenceEncoder.decodeFromBase64(s);
238         }
239     }
