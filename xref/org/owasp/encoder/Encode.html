42  /**
43   * Encode -- fluent interface for contextual encoding.  Example usage in a JSP:
44   *
45   * <pre>
46   *     &lt;input value="&lt;%=Encode.forHtml(value)%>" />
47   * </pre>
48   *
49   * <p>There are two versions of each contextual encoding method.  The first
50   * takes a {@code String} argument and returns the encoded version as a
51   * {@code String}.  The second version writes the encoded version directly
52   * to a {@code Writer}.</p>
53   *
54   * <p>Please make sure to read and understand the context that the method encodes
55   * for.  Encoding for the incorrect context will likely lead to exposing a
56   * cross-site scripting vulnerability.</p>
57   *
58   * @author Jeff Ichnowski
59   */
60  public final class Encode {
61      /** No instances. */
62      private Encode() {}
63  
64      /**
65       * <p>Encodes for (X)HTML text content and text attributes.  Since
66       * this method encodes for both contexts, it may be slightly less
67       * efficient to use this method over the methods targeted towards
68       * the specific contexts ({@link #forHtmlAttribute(String)} and
69       * {@link #forHtmlContent(String)}.  In general this method should
70       * be preferred unless you are really concerned with saving a few
71       * bytes or are writing a framework that utilizes this
72       * package.</p>
73       *
74       * <h5>Example JSP Usage</h5>
75       * <pre>
76       *     &lt;div>&lt;%=Encode.forHtml(unsafeData)%>&lt;/div>
77       *
78       *     &lt;input value="&lt;%=Encode.forHtml(unsafeData)%>" />
79       * </pre>
80       *
81       * <h5>Encoding Table</h5>
82       * <table border="0">
83       *   <thead>
84       *     <tr bgcolor="#ccf">
85       *       <th align="left">Input</th>
86       *       <th align="left">Result</th>
87       *     </tr>
88       *   </thead>
89       *   <tbody>
90       *     <tr>
91       *       <td>&ldquo;{@code &}&rdquo;</td>
92       *       <td>&ldquo;{@code &amp;}&rdquo;</td>
93       *     </tr>
94       *     <tr>
95       *       <td>&ldquo;{@code <}&rdquo;</td>
96       *       <td>&ldquo;{@code &lt;}&rdquo;</td>
97       *     </tr>
98       *     <tr>
99       *       <td>&ldquo;{@code >}&rdquo;</td>
100      *       <td>&ldquo;{@code &gt;}&rdquo;</td>
101      *     </tr>
102      *     <tr>
103      *       <td>&ldquo;{@code "}&rdquo;</td>
104      *       <td>&ldquo;{@code &#34;}&rdquo;</td>
105      *     </tr>
106      *     <tr>
107      *       <td>&ldquo;{@code '}&rdquo;</td>
108      *       <td>&ldquo;{@code &#39;}&rdquo;</td>
109      *     </tr>
110      *   </tbody>
111      * </table>
112      *
113      * <h5>Additional Notes</h5>
114      * <ul>
115      *
116      * <li>The encoding of the greater-than sign ({@code >}) is not
117      * strictly required, but is included for maximum
118      * compatibility.</li>
119      *
120      * <li>Numeric encoding is used for double-quote character ({@code
121      * "}) as it shorter than the also valid {@code &quot;}.</li>
122      *
123      * <li>Carriage return (U+0D), line-feed (U+0A), horizontal tab
124      * (U+09) and space (U+20) are valid in quoted attributes and in
125      * block in an unescaped form.</li>
126      *
127      * <li>Surrogate pairs are passed through only if valid.</li>
128      *
129      * <li>Characters that are not <a
130      * href="http://www.w3.org/TR/REC-xml/#charsets">valid according
131      * to the XML specification</a> are replaced by a space character
132      * as they could lead to parsing errors.  In particular only {@code #x9
133      * | #xA | #xD | [#x20-#xD7FF] | [#xE000-#xFFFD] |
134      * [#x10000-#x10FFFF]} are considered valid.</li>
135      *
136      * </ul>
137      *
138      * @param input the data to encode
139      * @return the data encoded for html.
140      */
141     public static String forHtml(String input) {
142         return forXml(input);
143     }

324     /**
325      * <p>Encodes for unquoted HTML attribute values.  {@link
326      * #forHtml(String)} or {@link #forHtmlAttribute(String)} should
327      * usually be preferred over this method as quoted attributes are
328      * XHTML compliant.</p>
329      *
330      * <p>When using this method, the caller is not required to
331      * provide quotes around the attribute (since it is encoded for
332      * such context).  The caller should make sure that the attribute
333      * value does not abut unsafe characters--and thus should usually
334      * err on the side of including a space character after the
335      * value.</p>
336      *
337      * <p>Use of this method is discouraged as quoted attributes are
338      * generally more compatible and safer.  Also note, that no
339      * attempt has been made to optimize this encoding, though it is
340      * still probably faster than other encoding libraries.</p>
341      *
342      * <h5>Example JSP Usage</h5>
343      * <pre>
344      *     &lt;input value=&lt;%=Encode.forHtmlUnquotedAttribute(input)%> >
345      * </pre>
346      *
347      * <h5>Encoding Table</h5>
348      * <table border="0">
349      *   <thead>
350      *     <tr bgcolor="#ccf">
351      *       <th align="left">Input</th>
352      *       <th align="left">Result</th>
353      *     </tr>
354      *   </thead>
355      *   <tbody>
356      *     <tr><td>{@code U+0009} (horizontal tab)</td><td>&ldquo;{@code &#9;}&rdquo;</td></tr>
357      *     <tr><td>{@code U+000A} (line feed)</td><td>&ldquo;{@code &#10;}&rdquo;</td></tr>
358      *     <tr><td>{@code U+000C} (form feed)</td><td>&ldquo;{@code &#12;}&rdquo;</td></tr>
359      *     <tr><td>{@code U+000D} (carriage return)</td><td>&ldquo;{@code &#13;}&rdquo;</td></tr>
360      *     <tr><td>{@code U+0020} (space)</td><td>&ldquo;{@code &#32;}&rdquo;</td></tr>
361      *     <tr><td>&ldquo;{@code &}&rdquo;</td><td>&ldquo;{@code &amp;}&rdquo;</td></tr>
362      *     <tr><td>&ldquo;{@code <}&rdquo;</td><td>&ldquo;{@code &lt;}&rdquo;</td></tr>
363      *     <tr><td>&ldquo;{@code >}&rdquo;</td><td>&ldquo;{@code &gt;}&rdquo;</td></tr>
364      *     <tr><td>&ldquo;{@code "}&rdquo;</td><td>&ldquo;{@code &#34;}&rdquo;</td></tr>
365      *     <tr><td>&ldquo;{@code '}&rdquo;</td><td>&ldquo;{@code &#39;}&rdquo;</td></tr>
366      *     <tr><td>&ldquo;{@code /}&rdquo;</td><td>&ldquo;{@code &#47;}&rdquo;</td></tr>
367      *     <tr><td>&ldquo;{@code =}&rdquo;</td><td>&ldquo;{@code &#61;}&rdquo;</td></tr>
368      *     <tr><td>&ldquo;{@code `}&rdquo;</td><td>&ldquo;{@code &#96;}&rdquo;</td></tr>
369      *     <tr><td>{@code U+0085} (next line)</td><td>&ldquo;{@code &#133;}&rdquo;</td></tr>
370      *     <tr><td>{@code U+2028} (line separator)</td><td>&ldquo;{@code &#8232;}&rdquo;</td></tr>
371      *     <tr><td>{@code U+2029} (paragraph separator)</td><td>&ldquo;{@code &#8233;}&rdquo;</td></tr>
372      *   </tbody>
373      * </table>
374      *
375      * <h5>Additional Notes</h5>
376      * <ul>
377      *
378      * <li>The following characters are <i>not</i> encoded:
379      * {@code 0-9, a-z, A-Z}, &ldquo;{@code !}&rdquo;, &ldquo;{@code
380      * #}&rdquo;, &ldquo;{@code $}&rdquo;, &ldquo;{@code %}&rdquo;,
381      * &ldquo;{@code (}&rdquo;, &ldquo;{@code )}&rdquo;, &ldquo;{@code
382      * *}&rdquo;, &ldquo;{@code +}&rdquo;, &ldquo;{@code ,}&rdquo;,
383      * &ldquo;{@code -}&rdquo;, &ldquo;{@code .}&rdquo;, &ldquo;{@code
384      * [}&rdquo;, &ldquo;{@code \}&rdquo;, &ldquo;{@code ]}&rdquo;,
385      * &ldquo;{@code ^}&rdquo;, &ldquo;{@code _}&rdquo;, &ldquo;{@code
386      * }}&rdquo;.</li>
387      *
388      * <li>Surrogate pairs are passed through only if valid.  Invalid
389      * surrogate pairs are replaced by a hyphen (-).</li>
390      *
391      * <li>Characters in the C0 and C1 control blocks and not
392      * otherwise listed above are considered invalid and replaced by a
393      * hyphen (-) character.</li>
394      *
395      * <li>Unicode "non-characters" are replaced by hyphens (-).</li>
396      *
397      * </ul>
398      *
399      * @param input the attribute value to be encoded.
400      * @return the attribute value encoded for unquoted attribute
401      * context.

434     /**
435      * Encodes for CSS strings.  The context must be surrounded by quotation
436      * characters.  It is safe for use in both style blocks and attributes in
437      * HTML.
438      *
439      * <h5>Example JSP Usage</h5>
440      * <pre>
441      *     &lt;div style="background: url('&lt;=Encode.forCssString(...)%>');">
442      *
443      *     &lt;style type="text/css">
444      *         background: url('&lt;%=Encode.forCssString(...)%>');
445      *     &lt;/style>
446      * </pre>
447      *
448      * <h5>Encoding  Notes</h5>
449      * <ul>
450      *
451      * <li>The following characters are encoded using hexidecimal
452      * encodings: {@code U+0000} - {@code U+001f},
453      * &ldquo;{@code "}&rdquo;,
454      * &ldquo;{@code '}&rdquo;,
455      * &ldquo;{@code \}&rdquo;,
456      * &ldquo;{@code <}&rdquo;,
457      * &ldquo;{@code &}&rdquo;,
458      * &ldquo;{@code (}&rdquo;,
459      * &ldquo;{@code )}&rdquo;,
460      * &ldquo;{@code /}&rdquo;,
461      * &ldquo;{@code >}&rdquo;,
462      * {@code U+007f},
463      * line separator ({@code U+2028}),
464      * paragraph separator ({@code U+2029}).</li>
465      *
466      * <li>Any character requiring encoding is encoded as {@code \xxx}
467      * where {@code xxx} is the shortest hexidecimal representation of
468      * its Unicode code point (after decoding surrogate pairs if
469      * necessary).  This encoding is never zero padded.  Thus, for
470      * example, the tab character is encoded as {@code \9}, not {@code
471      * \0009}.</li>
472      *
473      * <li>The encoder looks ahead 1 character in the input and
474      * appends a space to an encoding to avoid the next character
475      * becoming part of the hexidecimal encoded sequence.  Thus
476      * &ldquo;{@code '1}&rdquo; is encoded as &ldquo;{@code \27
477      * 1}&rdquo;, and not as &ldquo;{@code \271}&rdquo;.  If a space
478      * is not necessary, it is not included, thus &ldquo;{@code
479      * 'x}&rdquo; is encoded as &ldquo;{@code \27x}&rdquo;, and not as
480      * &ldquo;{@code \27 x}&rdquo;.</li>
481      *
482      * <li>Surrogate pairs are passed through only if valid.  Invalid
483      * surrogate pairs are replaced by an underscore (_).</li>
484      *
485      * <li>Unicode "non-characters" are replaced by underscores (_).</li>
486      *
487      * </ul>
488      *
489      * @param input the input to encode
490      * @return the encoded result
491      */
492     public static String forCssString(String input) {
493         // need to watch out for CSS expressions
494         return encode(Encoders.CSS_STRING_ENCODER, input);
495     }

1246     /**
1247      * A buffer used for encoding.
1248      */
1249     static class Buffer {
1250         /**
1251          * Input buffer size, used to extract a copy of the input
1252          * from a string and then send to the encoder.
1253          */
1254         static final int INPUT_BUFFER_SIZE = 1024;
1255         /**
1256          * Output buffer size used to store the encoded output before
1257          * wrapping in a string.
1258          */
1259         static final int OUTPUT_BUFFER_SIZE = INPUT_BUFFER_SIZE * 2;
1260 
1261         /**
1262          * The input buffer.  A heap-allocated, array-backed buffer of
1263          * INPUT_BUFFER_SIZE used for holding the characters to encode.
1264          */
1265         final CharBuffer _input = CharBuffer.allocate(INPUT_BUFFER_SIZE);
1266         /**
1267          * The output buffer.  A heap-allocated, array-backed buffer of
1268          * OUTPUT_BUFFER_SIZE used for holding the encoded output.
1269          */
1270         final CharBuffer _output = CharBuffer.allocate(OUTPUT_BUFFER_SIZE);
1271 
1272         /**
1273          * The core String encoding routine of this class.  It uses the input
1274          * and output buffers to allow the encoders to work in reuse arrays.
1275          * When the input and/or output exceeds the capacity of the reused
1276          * arrays, temporary ones are allocated and then discarded after
1277          * the encode is done.
1278          *
1279          * @param encoder the encoder to use
1280          * @param str the string to encode
1281          * @param j the offset in {@code str} to start encoding
1282          * @return the encoded result
1283          */
1284         String encode(Encoder encoder, String str, int j) {
1285             final int n = str.length();
1286             final int remaining = n - j;
1287 
1288             if (remaining <= INPUT_BUFFER_SIZE && j <= OUTPUT_BUFFER_SIZE) {
1289                 // the remaining input to encode fits completely in the pre-
1290                 // allocated buffer.
1291                 str.getChars(0, j, _output.array(), 0);
1292                 str.getChars(j, n, _input.array(), 0);
1293 
1294                 _input.limit(remaining).position(0);
1295                 _output.clear().position(j);
1296 
1297                 CoderResult cr = encoder.encodeArrays(_input, _output, true);
1298                 if (cr.isUnderflow()) {
1299                     return new String(_output.array(), 0, _output.position());
1300                 }
1301 
1302                 // else, it's an overflow, we need to use a new output buffer
1303                 // we'll allocate this buffer to be the exact size of the worst
1304                 // case, guaranteeing a second overflow would not be possible.
1305                 CharBuffer tmp = CharBuffer.allocate(
1306                     _output.position() +
1307                     encoder.maxEncodedLength(_input.remaining()));
1308 
1309                 // copy over everything that has been encoded so far
1310                 tmp.put(_output.array(), 0, _output.position());
1311 
1312                 cr = encoder.encodeArrays(_input, tmp, true);
1313                 if (cr.isOverflow()) {
1314                     throw new AssertionError("unexpected result from encoder");
1315                 }
1316 
1317                 return new String(tmp.array(), 0, tmp.position());
1318             } else {
1319                 // the input it too large for our pre-allocated buffers
1320                 // we'll use a temporary direct heap allocation
1321                 final int m = j + encoder.maxEncodedLength(remaining);
1322                 CharBuffer buffer = CharBuffer.allocate(m);
1323                 str.getChars(0, j, buffer.array(), 0);
1324                 str.getChars(j, n, buffer.array(), m - remaining);
1325 
1326                 CharBuffer input = buffer.duplicate();
1327                 input.limit(m).position(m-remaining);
1328                 buffer.position(j);
1329 
1330                 CoderResult cr = encoder.encodeArrays(input, buffer, true);
1331 
1332                 if (cr.isOverflow()) {
1333                     throw new AssertionError("unexpected result from encoder");
1334                 }
1335 
1336                 return new String(buffer.array(), 0, buffer.position());
1337             }
1338         }
1339 
1340         /**
1341          * The core Writer encoding routing of this class.  It uses the
1342          * input and output buffers to allow the encoders to reuse arrays.
1343          * Unlike the string version, this method will never allocate more
1344          * memory, instead encoding is done in batches and flushed to the
1345          * writer in batches as large as possible.
1346          *
1347          * @param encoder the encoder to use
1348          * @param out where to write the encoded output
1349          * @param str the string to encode
1350          * @param j the position in the string at which the first character
1351          * needs encoding.
1352          * @throws IOException if thrown by the writer.
1353          */
1354         void encode(Encoder encoder, Writer out, String str, int j)
1355             throws IOException
1356         {
1357             out.write(str, 0, j);
1358 
1359             final int n = str.length();
1360 
1361             _input.clear();
1362             _output.clear();
1363 
1364             final char[] inputArray = _input.array();
1365             final char[] outputArray = _output.array();
1366 
1367             for (;;) {
1368                 final int remainingInput = n - j;
1369                 final int startPosition = _input.position();
1370                 final int batchSize = Math.min(remainingInput, _input.remaining());
1371                 str.getChars(j, j+batchSize, inputArray, startPosition);
1372 
1373                 _input.limit(startPosition + batchSize);
1374 
1375 
1376                 for (;;) {
1377                     CoderResult cr = encoder.encodeArrays(
1378                         _input, _output, batchSize == remainingInput);
1379 
1380                     if (cr.isUnderflow()) {
1381                         // get next input batch
1382                         break;
1383                     }
1384 
1385                     // else, output buffer full, flush and continue.
1386                     out.write(outputArray, 0, _output.position());
1387                     _output.clear();
1388                 }
1389 
1390                 j += _input.position() - startPosition;
1391 
1392                 if (j == n) {
1393                     // done.  flush remaining output buffer and return
1394                     out.write(outputArray, 0, _output.position());
1395                     return;
1396                 }
1397 
1398                 _input.compact();
1399             }
1400         }
