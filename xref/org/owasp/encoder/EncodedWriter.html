34  package org.owasp.encoder;
35  
36  import java.io.IOException;
37  import java.io.Writer;
38  import java.nio.CharBuffer;
39  import java.nio.charset.CoderResult;
40  
41  /**
42   * EncodedWriter -- A writer the encodes all input for a specific context and writes the encoded output to another writer.
43   *
44   * @author Jeff Ichnowski
45   */
46  public class EncodedWriter extends Writer {
47  
48      /**
49       * Buffer size to allocate.
50       *
51       */
52      static final int BUFFER_SIZE = 1024;
53      /**
54       * Buffer to use for handling characters remaining in the input buffer after an encode. The value is set high enough to handle
55       * the lookaheads of all the encoders in the package.
56       */
57      static final int LEFT_OVER_BUFFER = 16;
58  
59      /**
60       * The wrapped writer.
61       */
62      private Writer _out;
63  
64      /**
65       * The encoder used to encode input to the output writer.
66       */
67      private Encoder _encoder;
68  
69      /**
70       * Where encoded output is buffered before sending on to the output writer.
71       */
72      private CharBuffer _buffer = CharBuffer.allocate(BUFFER_SIZE);
73  
74      /**
75       * Some encoders require more input or an explicit end-of-input flag before they will process the remaining characters of an
76       * input buffer. Because the writer API cannot pass this information on to the caller (e.g. by returning how many bytes were
77       * actually written), this writer implementation must buffer up the remaining characters between calls. The
78       * <code>_hasLeftOver</code> boolean is a flag used to indicate that there are left over characters in the buffer.
79       */
80      private boolean _hasLeftOver;
81  
82      /**
83       * See comment on _hasLeftOver. This buffer is created on-demand once. Whether it has anything to flush is determined by the
84       * _hasLeftOver flag.
85       */
86      private CharBuffer _leftOverBuffer;
87  
88      /**
89       * Creates an EncodedWriter that uses the specified encoder to encode all input before sending it to the wrapped writer.
90       *
91       * @param out the target for all writes
92       * @param encoder the encoder to use
93       */
94      public EncodedWriter(Writer out, Encoder encoder) {
95          super(out);
96  
97  //      Reduntant null check, super(out) checks for null and throws NPE.
98  //        if (out == null) {
99  //            throw new NullPointerException("writer must not be null");
100 //        }
101         if (encoder == null) {
102             throw new NullPointerException("encoder must not be null");
103         }
104 
105         _out = out;
106 
107         _encoder = encoder;
108     }
109 
110     /**
111      * Creates an EncodedWriter that uses the specified encoder to encode all input before sending it to the wrapped writer. This
112      * method is equivalent to calling:
113      * <pre>
114      *     new EncodedWriter(out, Encoders.forName(contextName));
115      * </pre>
116      *
117      * @param out the target for all writes
118      * @param contextName the encoding context name.
119      * @throws UnsupportedContextException if the contextName is unrecognized or not supported.
120      */
121     public EncodedWriter(Writer out, String contextName) throws UnsupportedContextException {
122         this(out, Encoders.forName(contextName));
123     }
124 
125     @Override
126     public void write(char[] cbuf, int off, int len) throws IOException {
127         synchronized (lock) {
128             CharBuffer input = CharBuffer.wrap(cbuf);
129             input.limit(off + len).position(off);
130 
131             flushLeftOver(input);
132 
133             for (;;) {
134                 CoderResult cr = _encoder.encode(input, _buffer, false);
135 
136                 if (cr.isUnderflow()) {
137                     if (input.hasRemaining()) {
138                         if (_leftOverBuffer == null) {
139                             _leftOverBuffer = CharBuffer.allocate(LEFT_OVER_BUFFER);
140                         }
141                         _leftOverBuffer.put(input);
142                         _hasLeftOver = true;
143                     }
144                     return;
145                 }
146                 if (cr.isOverflow()) {
147                     flushBufferToWriter();
148                 }
149             }
150         }
151     }
152 
153     /**
154      * Flushes the contents of the buffer to the writer and resets the buffer to make room for more input.
155      *
156      * @throws IOException thrown by the wrapped output.
157      */
158     private void flushBufferToWriter() throws IOException {
159         _out.write(_buffer.array(), 0, _buffer.position());
160         _buffer.clear();
161     }
162 
163     /**
164      * Flushes the left-over buffer. Characters from the input buffer are used to add more data to the _leftOverBuffer in order to
165      * make the flush happen.
166      *
167      * @param input the next input to encode, or null if at end of file.
168      * @throws IOException from the underlying writer.
169      */
170     private void flushLeftOver(CharBuffer input) throws IOException {
171         if (!_hasLeftOver) {
172             return;
173         }
174 
175         for (;;) {
176             if (input != null && input.hasRemaining()) {
177                 _leftOverBuffer.put(input.get());
178             }
179 
180             _leftOverBuffer.flip();
181             CoderResult cr = _encoder.encode(_leftOverBuffer, _buffer, input == null);
182 
183             if (cr.isUnderflow()) {
184                 if (_leftOverBuffer.hasRemaining()) {
185                     _leftOverBuffer.compact();
186                 } else {
187                     break;
188                 }
189             }
190             if (cr.isOverflow()) {
191                 flushBufferToWriter();
192             }
193         }
194 
195         _hasLeftOver = false;
196         _leftOverBuffer.clear();
197     }
198 
199     @Override
200     public void flush() throws IOException {
201         synchronized (lock) {
202             flushBufferToWriter();
203             _out.flush();
204         }
205     }
206 
207     @Override
208     public void close() throws IOException {
209         synchronized (lock) {
210             flushLeftOver(null);
211             flushBufferToWriter();
212             _out.close();
213         }
214     }
215 }
