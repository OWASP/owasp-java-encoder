35  package org.owasp.encoder;
36  
37  import junit.framework.Test;
38  import junit.framework.TestCase;
39  import junit.framework.TestSuite;
40  
41  /**
42   * URIEncoderTest -- test suite for the URIEncoder.
43   *
44   * @author Jeff Ichnowski
45   */
46  public class URIEncoderTest extends TestCase {
47      public static Test suite() {
48          TestSuite suite = new TestSuite();
49          for (URIEncoder.Mode mode : URIEncoder.Mode.values()) {
50              EncoderTestSuiteBuilder builder = new EncoderTestSuiteBuilder(new URIEncoder(mode), "-safe-", "<>")
51                  .encoded(0, Character.MAX_CODE_POINT)
52                  .invalid(Character.MIN_SURROGATE, Character.MAX_SURROGATE)
53  
54                  // Characters safe in all modes
55                  .valid("-._~")
56                  .valid('a', 'z')
57                  .valid('A', 'Z')
58                  .valid('0', '9');
59  
60              switch (mode) {
61              case COMPONENT:
62                  // reserved gen-delims
63                  builder.encode("%3A", ":");
64                  builder.encode("%2F", "/");
65                  builder.encode("%3F", "?");
66                  builder.encode("%23", "#");
67                  builder.encode("%5B", "[");
68                  builder.encode("%5D", "]");
69                  builder.encode("%40", "@");
70  
71                  // reserved sub-delims
72                  builder.encode("%21", "!");
73                  builder.encode("%24", "$");
74                  builder.encode("%26", "&");
75                  builder.encode("%27", "'");
76                  builder.encode("%28", "(");
77                  builder.encode("%29", ")");
78                  builder.encode("%2A", "*");
79                  builder.encode("%2B", "+");
80                  builder.encode("%2C", ",");
81                  builder.encode("%3B", ";");
82                  builder.encode("%3D", "=");
83                  break;
84              case FULL_URI:
85                  builder.encode(
86                      "full-url",
87                      "http://www.owasp.org/index.php?foo=bar&baz#fragment",
88                      "http://www.owasp.org/index.php?foo=bar&baz#fragment");
89                  builder.valid(":/?#[]@!$&'()*+,;=");
90                  break;
91              default:
92                  throw new AssertionError("untested mode: "+mode);
93              }
94              suite.addTest(builder
95                  // simple test of some unencoded characters
96                  .encode("abcABC123", "abcABC123")
97  
98                  // ASCII characters encoded in all modes
99                  .encode("%20", " ")
100                 .encode("%22", "\"")
101                 .encode("%25", "%")
102                 .encode("%3C", "<")
103                 .encode("%3E", ">")
104                 .encode("%5C", "\\")
105                 .encode("%5E", "^")
106                 .encode("%60", "`")
107                 .encode("%7B", "{")
108                 .encode("%7C", "|")
109                 .encode("%7D", "}")
110 
111                 // UTF-8 multi-byte handling
112                 .encode("2-byte-utf-8", "%C2%A0", "\u00a0")
113                 .encode("3-byte-utf-8", "%E0%A0%80", "\u0800")
114                     // TODO: double check that this is how the surrogate pair should
115                     // be encoded
116                 .encode("4-byte-utf-8", "%F0%90%80%80",
117                     new StringBuilder().appendCodePoint(0x10000).toString())
118 
119                 .encode("missing-low-surrogate", "-", "\ud800")
120                 .encode("missing-high-surrogate", "-", "\udc00")
121 
122                 .validSuite()
123                 .invalidSuite('-')
124                 .encodedSuite()
125 
126                 .build());
127         }
128         return suite;
129     }
130 }
