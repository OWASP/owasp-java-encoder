35  package org.owasp.encoder;
36  
37  import java.nio.CharBuffer;
38  import java.nio.charset.CoderResult;
39  
40  /**
41   * <p>HTMLEncoder -- an encoder for HTML contexts.  Currently most
42   * HTML-based contexts are properly handled by {@line XMLEncoder}.
43   * The remaining HTML-specific context of "unquoted attributes" could
44   * not be added to the XMLEncoder without slowing it down.  This class
45   * implements that remaining context: <strong>unquoted attribute
46   * values</strong>.</p>
47   *
48   * <p>Note: because this context is likely small strings, and
49   * hopefully rarely used, no effort was put into optimizing this
50   * encoder.</p>
51   *
52   * @author Jeff Ichnowski
53   */
54  class HTMLEncoder extends Encoder {
55      /**
56       * Number of characters in the encoding prefix and suffix when using
57       * decimal numeric encodings of the form "&#...;".
58       */
59      private static final int ENCODE_AFFIX_CHAR_COUNT = 3;
60  
61      /** Encoding for '\t'. */
62      private static final char[] TAB = "&#9;".toCharArray();
63      /** Encoding for '&amp;'. */
64      private static final char[] AMP = "&amp;".toCharArray();
65      /** Encoding for '&lt;'. */
66      private static final char[] LT = "&lt;".toCharArray();
67      /** Encoding for '&gt;'. */
68      private static final char[] GT = "&gt;".toCharArray();
69  
70      // The large table-switch implementation used here is fast to
71      // implement but slower at runtime than tuned-for-expected-input
72      // encoders that use selective if/else's.  Look at the results of
73      // BenchmarkTest to see the difference.  See note in javadoc as to
74      // reasoning.
75  
76      // On Core i7 (Sandybridge)
77      // Baseline is 371.401009 ns/op
78      // Benchmarked Encode.forXml: 324.219992 ns/op (-12.70% on baseline)
79      // Benchmarked Encode.forHtmlUnquotedAttribute: 821.583263 ns/op (+121.21% on baseline)
80  
81  
82      @Override
83      int maxEncodedLength(int n) {
84          // if everything is line separators and paragraph separators then
85          // we get "&#8283;"
86          return n*(ENCODE_AFFIX_CHAR_COUNT+4);
87      }
88  
89      @Override
90      int firstEncodedOffset(String input, int off, int len) {
91          final int n = off+len;
92          for (int i=off ; i<n ; ++i) {
93              final char ch = input.charAt(i);
94  
95              switch (ch) {
96              case '\t': case '\r': case '\f': case '\n': case ' ': case Unicode.NEL:
97              case '\"': case '\'':
98              case '/': case '=': case '`':
99              case '&': case '<': case '>':
100                 return i;
101 
102             case '!': case '#': case '$': case '%':
103             case '(': case ')': case '*': case '+':
104             case ',': case '-': case '.':
105 
106             case '0': case '1': case '2': case '3': case '4':
107             case '5': case '6': case '7': case '8': case '9':
108             case ':': case ';': case '?': case '@':
109 
110             case 'A': case 'B': case 'C': case 'D': case 'E':
111             case 'F': case 'G': case 'H': case 'I': case 'J':
112             case 'K': case 'L': case 'M': case 'N': case 'O':
113             case 'P': case 'Q': case 'R': case 'S': case 'T':
114             case 'U': case 'V': case 'W': case 'X': case 'Y':
115             case 'Z':
116 
117             case '[': case '\\': case ']': case '^': case '_':
118 
119             case 'a': case 'b': case 'c': case 'd': case 'e':
120             case 'f': case 'g': case 'h': case 'i': case 'j':
121             case 'k': case 'l': case 'm': case 'n': case 'o':
122             case 'p': case 'q': case 'r': case 's': case 't':
123             case 'u': case 'v': case 'w': case 'x': case 'y':
124             case 'z':
125 
126             case '{': case '|': case '}': case '~':
127                 break; // valid
128 
129             default:
130 
131                 if (Character.isHighSurrogate(ch)) {
132                     if (i+1 < n) {
133                         if (Character.isLowSurrogate(input.charAt(i+1))) {
134                             int cp = Character.toCodePoint(ch, input.charAt(i+1));
135                             if (Unicode.isNonCharacter(cp)) {
136                                 return i;
137                             } else {
138                                 ++i;
139                             }
140                             break;
141                         }
142                     } else {
143                         return i;
144                     }
145                 }
146 
147                 if (ch <= Unicode.MAX_C1_CTRL_CHAR ||
148                     Character.MIN_SURROGATE <= ch && ch <= Character.MAX_SURROGATE ||
149                     ch > '\ufffd' ||
150                     ('\ufdd0' <= ch && ch <= '\ufdef') ||
151                     ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR)
152                 {
153                     return i;
154                 }
155             }
156         }
157         return n;
158     }
159 
160     /**
161      * Appends a source array verbatim to the output array.  Caller must
162      * insure there is enough space in the array for the output.
163      *
164      * @param src the characters to copy
165      * @param out the output buffer
166      * @param j the offset where to write in the output buffer
167      * @return {@code j + src.length}
168      */
169     static int append(char[] src, char[] out, int j) {
170         System.arraycopy(src, 0, out, j, src.length);
171         return j + src.length;
172     }
173 
174     /**
175      * Appends the numerically encoded version of {@code codePoint} to the
176      * output buffer. Caller must insure there is enough space for the
177      * output.
178      *
179      * @param codePoint the character to encode
180      * @param out the output buffer
181      * @param j the offset where to write in the output buffer
182      * @return {@code j} + the encoded length.
183      */
184     static int encode(int codePoint, char[] out, int j) {
185         out[j++] = '&';
186         out[j++] = '#';
187         if (codePoint >= 1000) {
188             out[j++] = (char)(codePoint / 1000 % 10 + '0');
189         }
190         if (codePoint >= 100) {
191             out[j++] = (char)(codePoint / 100 % 10 + '0');
192         }
193         if (codePoint >= 10) {
194             out[j++] = (char)(codePoint / 10 % 10 + '0');
195         }
196         out[j++] = (char)(codePoint % 10 + '0');
197         out[j++] = ';';
198         return j;
199     }
200 
201     @Override
202     CoderResult encodeArrays(
203         CharBuffer input, CharBuffer output, boolean endOfInput)
204     {
205         final char[] in = input.array();
206         final char[] out = output.array();
207         int i = input.arrayOffset() + input.position();
208         final int n = input.arrayOffset() + input.limit();
209         int j = output.arrayOffset() + output.position();
210         final int m = output.arrayOffset() + output.limit();
211 
212     charLoop:
213         for ( ; i<n ; ++i) {
214             final char ch = in[i];
215 
216             // gigantic switch, hopefully compiled to a tableswitch.
217             // this approach appears to be slower than the if/else
218             // approach used in the other encoders.  Perhaps an artifact
219             // of the CPU's branch predictor, or possible additional
220             // overhead of range checking, or having the entire table
221             // available to the cache.  If time allows, it would
222             // interesting to find out.
223             switch (ch) {
224             case '\t':
225                 if (j + TAB.length > m) {
226                     return overflow(input, i, output, j);
227                 }
228                 j = append(TAB, out, j);
229                 break;
230 
231             case '\r': case '\n': case '\f': case ' ': case '\"': case '\'':
232             case '/': case '=': case '`':
233                 if (ENCODE_AFFIX_CHAR_COUNT+2+j > m) {
234                     return overflow(input, i, output, j);
235                 }
236                 j = encode(ch, out, j);
237                 break;
238 
239             case Unicode.NEL:
240                 if (ENCODE_AFFIX_CHAR_COUNT+3+j > m) {
241                     return overflow(input, i, output, j);
242                 }
243                 j = encode(ch, out, j);
244                 break;
245 
246             case '&':
247                 if (j + AMP.length > m) {
248                     return overflow(input, i, output, j);
249                 }
250                 j = append(AMP, out, j);
251                 break;
252 
253             case '<':
254                 if (j + LT.length > m) {
255                     return overflow(input, i, output, j);
256                 }
257                 j = append(LT, out, j);
258                 break;
259 
260             case '>':
261                 if (j + GT.length > m) {
262                     return overflow(input, i, output, j);
263                 }
264                 j = append(GT, out, j);
265                 break;
266 
267             case '!': case '#': case '$': case '%':
268             case '(': case ')': case '*': case '+':
269             case ',': case '-': case '.':
270 
271             case '0': case '1': case '2': case '3': case '4':
272             case '5': case '6': case '7': case '8': case '9':
273             case ':': case ';': case '?': case '@':
274 
275             case 'A': case 'B': case 'C': case 'D': case 'E':
276             case 'F': case 'G': case 'H': case 'I': case 'J':
277             case 'K': case 'L': case 'M': case 'N': case 'O':
278             case 'P': case 'Q': case 'R': case 'S': case 'T':
279             case 'U': case 'V': case 'W': case 'X': case 'Y':
280             case 'Z':
281 
282             case '[': case '\\': case ']': case '^': case '_':
283 
284             case 'a': case 'b': case 'c': case 'd': case 'e':
285             case 'f': case 'g': case 'h': case 'i': case 'j':
286             case 'k': case 'l': case 'm': case 'n': case 'o':
287             case 'p': case 'q': case 'r': case 's': case 't':
288             case 'u': case 'v': case 'w': case 'x': case 'y':
289             case 'z': case '{': case '|': case '}': case '~':
290                 if (j >= m) {
291                     return overflow(input, i, output, j);
292                 }
293                 out[j++] = ch;
294                 break;
295             default:
296 
297                 if (Character.isHighSurrogate(ch)) {
298                     if (i+1 < n) {
299                         if (Character.isLowSurrogate(in[i+1])) {
300                             int cp = Character.toCodePoint(ch, in[i+1]);
301                             if (Unicode.isNonCharacter(cp)) {
302                                 if (j >= m) {
303                                     return overflow(input, i, output, j);
304                                 }
305                                 out[j++] = '-';
306                                 ++i;
307                             } else {
308                                 if (j+1 >= m) {
309                                     return overflow(input, i, output, j);
310                                 }
311                                 out[j++] = ch;
312                                 out[j++] = in[++i];
313                             }
314                             break;
315                         }
316                     } else if (!endOfInput) {
317                         break charLoop;
318                     }
319                 }
320 
321                 if (j >= m) {
322                     return overflow(input, i, output, j);
323                 }
324 
325                 if (ch <= Unicode.MAX_C1_CTRL_CHAR ||
326                     Character.MIN_SURROGATE <= ch && ch <= Character.MAX_SURROGATE ||
327                     ch > '\ufffd' ||
328                     ('\ufdd0' <= ch && ch <= '\ufdef'))
329                 {
330                     // invalid
331                     out[j++] = '-';
332                 } else if (ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR) {
333                     if (ENCODE_AFFIX_CHAR_COUNT+4+j > m) {
334                         return overflow(input, i, output, j);
335                     }
336                     j = encode(ch, out, j);
337                 } else {
338                     out[j++] = ch;
339                 }
340             }
341         }
342 
343         return underflow(input, i, output, j);
344     }
