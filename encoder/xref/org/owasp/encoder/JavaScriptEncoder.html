35  package org.owasp.encoder;
36  
37  import java.nio.CharBuffer;
38  import java.nio.charset.CoderResult;
39  
40  /**
41   * JavaScriptEncoder -- An encoder for JavaScript string contexts.
42   *
43   * @author jeffi
44   */
45  class JavaScriptEncoder extends Encoder {
46  
47      /**
48       * Mode of operation constants for the JavaScriptEncoder.
49       */
50      static enum Mode {
51          /**
52           * Standard encoding of JavaScript Strings.  Escape sequences are chosen
53           * according to what is the shortest sequence possible for the character.
54           */
55          SOURCE,
56  
57          /**
58           * Encoding for use in HTML attributes.  Quote characters are escaped
59           * using hex encodes instead of backslashes.  The alternate would be
60           * to use a sequence of encodes that would actually be longer.  In this
61           * mode double-quote is "\x22" and single-quote is "\x27".  (In HTML
62           * attributes the alternate would be encoding "\"" and "\'" with entity
63           * escapes to "\&amp;#34;" and "\&amp;39;").
64           */
65          ATTRIBUTE,
66  
67          /**
68           * Encoding for use in HTML script blocks.  The main concern here is
69           * permaturely terminating a script block with a closing "&lt;/" inside
70           * the string.  This encoding escapes "/" as "\/" to prevent such
71           * termination.
72           */
73          BLOCK,
74  
75          /**
76           * Encodes for use in either HTML script attributes or blocks.
77           * Essentially this is both special escapes from HTML_ATTRIBUTE and
78           * HTML_CONTENT combined.
79           */
80          HTML,
81          ;
82      }
83  
84      /**
85       * The mode of operations--used for toString implementation.
86       */
87      private final Mode _mode;
88      /**
89       * True if quotation characters should be hex encoded.  Hex encoding
90       * quotes allows JavaScript to be included in XML attributes without
91       * additional XML-based encoding.
92       */
93      private final boolean _hexEncodeQuotes;
94      /**
95       * An array of 4 32-bit integers used as bitmasks to check if a character
96       * needs encoding or not.  If the bit is set, the character is valid and
97       * does not need encoding.
98       */
99      private final int[] _validMasks;
100     /**
101      * True if the output should only include ASCII characters.  Valid
102      * non-ASCII characters that would normally not be encoded, will be
103      * encoded.
104      */
105     private final boolean _asciiOnly;
106 
107     /**
108      * Constructs a new JavaScriptEncoder for the specified contextual mode.
109      *
110      * @param mode the mode of operation
111      * @param asciiOnly true if only ASCII characters should be included
112      * in the output (all code-points outside the ASCII range will be
113      * encoded).
114      */
115     JavaScriptEncoder(Mode mode, boolean asciiOnly) {
116         // TODO: after some testing it appears that an array of int masks
117         // is faster than two longs, or an array of longs or an array of bytes
118         // the other encoders based upon masks should be switched to ints.
119         // (to be clear, it's much faster on 32-bit VMS, and just slightly
120         // faster on 64-bit VMS)
121         _mode = mode;
122 
123         // Note: this probably needs to be repeated everywhere this trick is
124         // used, but here seems like as good a place as any.  According to
125         // the Java spec (x << y) where x and y are integers, is evaluated
126         // as (x << (y & 31)).  Or put another way, only the lower 5 bits
127         // of the shift amount are considered.
128         _validMasks = new int[] {
129             0,
130             -1 & ~((1 << '\'') | (1 << '\"')),
131             -1 & ~((1 << '\\')),
132             asciiOnly ? ~(1 << Unicode.DEL) : -1,
133         };
134 
135         if (mode == Mode.BLOCK || mode == Mode.HTML) {
136             // in <script> blocks, we need to prevent the browser from seeing
137             // "</anything>" and "<!--". To do so we escape "/" as "\/" and
138             // escape "-" as "\-".  Both could be solved with a hex encoding
139             // on "<" but we figure "<" appears often in script strings and
140             // the backslash encoding is more readable than a hex encoding.
141             // (And note, a backslash encoding would not prevent the exploits
142             // on "</...>" and "<!--".
143             // In short "</script>" is escaped as "<\/script>" and "<!--" is
144             // escaped as "<!\-\-".
145             _validMasks[1] &= ~((1 << '/') | (1 << '-'));
146         }
147         if (mode != Mode.SOURCE) {
148             _validMasks[1] &= ~(1 << '&');
149         }
150 
151         _asciiOnly = asciiOnly;
152         _hexEncodeQuotes = (mode == Mode.ATTRIBUTE || mode == Mode.HTML);
153     }
154 
155     @Override
156     int maxEncodedLength(int n) {
157         return n*6;
158     }
159 
160     @Override
161     int firstEncodedOffset(String input, int off, int len) {
162         final int n = off + len;
163         final int[] validMasks = this._validMasks;
164         for (int i=off ; i<n ; ++i) {
165             char ch = input.charAt(i);
166             if (ch < 128) {
167                 if ((validMasks[ch>>>5] & (1 << ch)) == 0) {
168                     return i;
169                 }
170             } else if (_asciiOnly || ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR) {
171                 return i;
172             }
173         }
174         return n;
175     }
176 
177     @Override
178     protected CoderResult encodeArrays(CharBuffer input, CharBuffer output, boolean endOfInput) {
179         final char[] in = input.array();
180         final char[] out = output.array();
181         int i = input.arrayOffset() + input.position();
182         final int n = input.arrayOffset() + input.limit();
183         int j = output.arrayOffset() + output.position();
184         final int m = output.arrayOffset() + output.limit();
185 
186         final int[] validMasks = this._validMasks;
187 
188         for ( ; i<n ; ++i) {
189             char ch = in[i];
190 
191         hexEncoded:
192             {
193             encoded:
194                 {
195                     if (ch < 128) {
196                         if ((validMasks[ch>>>5] & (1 << ch)) == 0) {
197                             break encoded;
198                         }
199                     } else if (_asciiOnly || ch == Unicode.LINE_SEPARATOR || ch == Unicode.PARAGRAPH_SEPARATOR) {
200                         if (ch <= 0xff) {
201                             break hexEncoded;
202                         }
203                         if (j+6 > m) {
204                             return overflow(input, i, output, j);
205                         }
206                         out[j++] = '\\';
207                         out[j++] = 'u';
208                         out[j++] = HEX[ch >>> 3 * HEX_SHIFT];
209                         out[j++] = HEX[ch >>> 2 * HEX_SHIFT & HEX_MASK];
210                         out[j++] = HEX[ch >>> HEX_SHIFT & HEX_MASK];
211                         out[j++] = HEX[ch & HEX_MASK];
212                         continue;
213                     }
214                     if (j >= m) {
215                         return overflow(input, i, output, j);
216                     }
217                     out[j++] = ch;
218                     continue;
219                 }
220 
221                 switch (ch) {
222                 case '\b':
223                     if (j+2 > m) {
224                         return overflow(input, i, output, j);
225                     }
226                     out[j++] = '\\';
227                     out[j++] = 'b';
228                     continue;
229                 case '\t':
230                     if (j+2 > m) {
231                         return overflow(input, i, output, j);
232                     }
233                     out[j++] = '\\';
234                     out[j++] = 't';
235                     continue;
236                 case '\n':
237                     if (j+2 > m) {
238                         return overflow(input, i, output, j);
239                     }
240                     out[j++] = '\\';
241                     out[j++] = 'n';
242                     continue;
243                 // Per Mike Samuel "\v should not be used since some
244                 // versions of IE treat it as a literal letter 'v'"
245 //                case 0x0b: // '\v'
246 //                    if (j+1 >= m) {
247 //                        return overflow(input, i, output, j);
248 //                    }
249 //                    out[j++] = '\\';
250 //                    out[j++] = 'v';
251 //                    break;
252                 case '\f':
253                     if (j+2 > m) {
254                         return overflow(input, i, output, j);
255                     }
256                     out[j++] = '\\';
257                     out[j++] = 'f';
258                     continue;
259                 case '\r':
260                     if (j+2 > m) {
261                         return overflow(input, i, output, j);
262                     }
263                     out[j++] = '\\';
264                     out[j++] = 'r';
265                     continue;
266                 case '\'':
267                 case '\"':
268                     if (_hexEncodeQuotes) {
269                         break hexEncoded;
270                     }
271                     // fall through
272                 case '\\':
273                 case '/':
274                 case '-':
275                     // We'll only see '/' and '-' here in the BLOCK and HTML
276                     // modes otherwise it will be accepted as valid by the
277                     // bitmasks.
278                     if (j+2 > m) {
279                         return overflow(input, i, output, j);
280                     }
281                     out[j++] = '\\';
282                     out[j++] = ch;
283                     continue;
284                 default:
285                     break;
286                 }
287             }
288 
289             if (j+4 > m) {
290                 return overflow(input, i, output, j);
291             }
292             out[j++] = '\\';
293             out[j++] = 'x';
294             out[j++] = HEX[ch >>> HEX_SHIFT];
295             out[j++] = HEX[ch & HEX_MASK];
296         }
297 
298         return underflow(input, i, output, j);
299     }
300 
301     @Override
302     public String toString() {
303         return "JavaScriptEncoder(mode="+_mode+","+(_asciiOnly?"ASCII":"UNICODE")+")";
304     }
